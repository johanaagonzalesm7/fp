Algoritmo NombresApellidos;
VAR
	Caracter N;
	Caracter Ap;
	Caracter Am;
Inicio 
	Escribir ("¿Cual es tu nombre(s):");
	Leer N;
	Escribir ("¿Cual es tu primer apellido:");

	Escribir ("¿Cual es tu segundo apellido:");
	Leer Am;
	Escribir ("Tu nombre completo es:", N, Ap, Am);
Fin
