Algoritmo Promedio;
VAR
	Int C1;
	Int C2;
	Int C3;
	Int C4;
	Int C5;
	Int Promedio;
Inicio 
	Escribir ("Primer calificacion:");
	Leer C1;
	Escribir ("Segundo calificacion:");
	Leer C2;
	Escribir ("Tercera calificacion:");
	Leer C3;
	Escribir ("Cuarta calificacion:");
	Leer C4;
	Escribir ("Quinta calificacion:");

	Escribir ("El promedio es:", Promedio);
